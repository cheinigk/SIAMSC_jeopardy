        \FOR {$k = 1 \cdots n$}
        \FOR {$i = k+1 \cdots n$}
        \STATE $m := A[i, k] / A[k, k]$
        \FOR {$j = k+1 \cdots n$}
        \STATE $A[i, j]  := A[i, j] - A[k, j] * m$
      \ENDFOR
      \STATE $A[i, k]  := 0$
    \ENDFOR
  \ENDFOR
\end{algorithmic}
\end{algorithm}
