% Define an environment for the Question Header and Footer

    fontlower=\LARGE\bfseries,
    height=.176\paperheight,
    sidebyside,
    segmentation style={white,solid,line width=.4pt},
    #1
}

% Command for setting up the header of a question
\newcommand{\header}[2]{% Gets 2 arguments: row, i.e. category and column, i.e. price
    \begin{QuestionHeadFoot}
        #1 \tcblower #2
    \end{QuestionHeadFoot}
}

% Command for setting up the footer of a question
\newcommand{\footer}[1]{% 1 arg: Question/Answer ID ?
    \begin{QuestionHeadFoot}[before=\vskip-2.7ex]
        \hyperlink{question#1}{Question} \\ \hyperlink{answer#1}{Answer} \tcblower
        \hideocg{#1}{Done!} \\ \hyperlink{home}{Home}
    \end{QuestionHeadFoot}
}

% Command for setting up the Question and the answer
\newcommand{\content}[4]{% 4 args: 1-buttonlinks link 2-categorynumber 3-price 4-text
    \begin{frame}
        \hypertarget<1>{answer#1}{}
        \hypertarget<2>{question#1}{}
        \hypertarget{#1}{}
        \header{#2}{#3}
        #4
        \footer{#1}
    \end{frame}
}


% Command for setting up the footer of a DOUBLE question
\newcommand{\footerdouble}[1]{
    \begin{QuestionHeadFoot}[before=\vskip-2.7ex]
        \hyperlink{question#1}{\color{red}Question} \\
        \hyperlink{answer#1}{\color{green}Answer} \tcblower
        \hideocg{#1}{Done!} \\
        \hyperlink{home}{Home}
    \end{QuestionHeadFoot}
}

% Command for setting up the DOUBLE question and answer
\newcommand{\contentdouble}[4]{%
	{
    \begin{frame}
        \hypertarget<1>{double#1}{}
        \hypertarget<3>{answer#1}{}
        \hypertarget<2>{question#1}{}
        \hypertarget{#1}{}
        \header{#2}{DOUBLE!!!!!}
        #4

        \footerdouble{#1}
    \end{frame}
  }
}
